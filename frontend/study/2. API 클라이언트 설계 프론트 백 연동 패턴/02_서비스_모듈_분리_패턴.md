# 02. 서비스 모듈 분리 패턴

> **현재 상태**: `apiClient.ts` + `reviewService.ts` 2개만 존재  
> 나머지 도메인은 훅(useRankings.ts 등)에서 `api.get()` 직접 호출 중

---

## 왜 서비스 레이어가 필요한가?

```
현재 (서비스 레이어 없음)            개선 후
─────────────────────────           ──────────────────────────
useRankings.ts                      useRankings.ts
  api.get('/rankings/global')  →      getGlobalRanking()   ← rankingService.ts
  api.get('/rankings/region?...') →   getRegionRanking(name)
  api.get('/rankings/health')  →      getHealthRanking()

문제: URL 문자열이 훅에 하드코딩        장점: URL 변경 시 서비스 파일 1곳만 수정
     타입 캐스팅 직접 관리              타입 안전성 보장 (반환 타입 명시)
     여러 곳에서 중복 경로 작성          재사용 가능
```

## 서비스 파일 작성 패턴 (rankingService.ts 예시)

```typescript
// frontend/src/services/rankingService.ts
import { api } from './apiClient';

// ── 타입: useRankings.ts에 분산된 인터페이스를 서비스 파일로 이동 ──
export interface UserRankResponse {
  userId: number;
  nickname: string;
  titleName: string;
  level: number;
  score: number;
  rank: number;
  equippedAvatarUrl?: string | null;
  equippedItems: { icon: string; name: string; type: 'AVATAR' | 'EFFECT'; iconType: 'URL' | 'DICEBEAR' | 'EMOJI' }[];
}

export interface RankingResponse {
  topRankers: UserRankResponse[];
  myRank: UserRankResponse;
  activeUserCount: number;
}

// ── API 함수: URL 조립은 여기서만, 호출부는 함수명만 알면 됨 ─────────
export const getGlobalRanking = () =>
  api.get<RankingResponse>('/rankings/global');

export const getRegionRanking = (regionName: string) =>
  api.get<RankingResponse>(`/rankings/region?regionName=${encodeURIComponent(regionName)}`);

export const getHealthRanking = () =>
  api.get<RankingResponse>('/rankings/health');
```

## notificationService.ts 핵심 포인트

```typescript
// frontend/src/services/notificationService.ts
import { api } from './apiClient';
import { NotificationDto } from '../types/api';

// SSE는 EventSource API를 써서 Authorization 헤더 전송 불가
// → 백엔드가 단기 SSE 전용 토큰을 별도 발급 (POST /sse-token)
export const getSseToken = () =>
  api.post<{ sseToken: string }>('/notifications/sse-token');

// EventSource에 직접 넣을 URL 반환 (api.get이 아닌 이유: 스트리밍)
export const getSubscribeUrl = (sseToken: string) =>
  `${import.meta.env.VITE_API_URL || ''}/api/v1/notifications/subscribe?sseToken=${sseToken}`;

export const getNotifications = () =>
  api.get<NotificationDto[]>('/notifications');

export const markNotificationRead = (id: number) =>
  api.patch<void>(`/notifications/${id}/read`);

export const markAllNotificationsRead = () =>
  api.post<void>('/notifications/mark-all-read');

export const deleteNotification = (id: number) =>
  api.delete<void>(`/notifications/${id}`);
```

## 도입 전후 코드 비교

```typescript
// ── Before: hooks/useRankings.ts ─────────────────────────────────
// URL이 훅에 직접 노출 → 경로 변경 시 훅 수정 필요
const res = await api.get('/rankings/global') as RankingResponse;

// ── After: hooks/useRankings.ts ──────────────────────────────────
import { getGlobalRanking } from '../services/rankingService';

const res = await getGlobalRanking(); // 타입 자동 추론, URL 불필요
```
